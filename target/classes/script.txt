fib(n) -> {
    if (n <= 2)
        return 1;
    return fib(n - 1) + fib(n - 2);
}
gib(n) -> {
    if (n <= 3)
        return 1;
    return fib(n - 3) + fib(n - 2) + fib(n - 1) ** 2 - (4 * 8 + 2);
}
rtp() -> {
    sum = 0;
    for (i = 0; i < 10; i = i + 1)
        sum = sum + fib(i) + gib(i);
    trb(n) -> gib(n) ** fib(n);
    return trb(5) + sum;
}
n = 1;
while (n) {
    print(n + rtp());
    for (i = 0; i < 100; i = i + 1)
        do_while(n)
                n = 0;
}
print(fib(30));
